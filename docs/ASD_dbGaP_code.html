anno1=tibble(ID=paste(annotation$locus, annotation1$ref, annotation1$alt, sep="_"), Annotation=annotation1$category, ExacAF=annotation1$exac)

ID=paste("chr", annotation2$V1[-1], ":", sep="")
start_end=paste(annotation2$V2[-1], "_", annotation2$V3[-1], "_", annotation2$V4[-1], "_", annotation2$V5[-1], sep="")
